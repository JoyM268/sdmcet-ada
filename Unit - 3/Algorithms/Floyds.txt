Floyd(W[1..n, 1..n])
Input: The weight matrix W of a graph with no negative-length cycle
Output: The distance matrix of the shortest paths lengths
D <- W
for k <- 1 to n do
    for i <- 1 to n do
        for j <- 1 to n do
            D[i, j] <- min{D[i,j], D[i, k]+ D[k,j]}
return D
